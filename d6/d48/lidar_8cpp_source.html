22 // Récupérer la pose absolue du robot
23 double rayStartX = simulation->getXRobotStart() + simulation->getRobot()->getX();
24 double rayStartY = simulation->getYRobotStart() + simulation->getRobot()->getY();
25 double orientation = simulation->getOrientationRobotStart() + simulation->getRobot()->getOrientation();

28 double rayAngle = orientation + (rayID - rayCount / 2) * (M_PI / 180.0);

31 double rayDirX = cos(rayAngle);
32 double rayDirY = sin(rayAngle);
33
34 // Longueur du rayon pour un déplacement unitaire en x ou y
35 double rayUnitStepSizeX = sqrt(1 + (rayDirY / rayDirX) * (rayDirY / rayDirX));
36 double rayUnitStepSizeY = sqrt(1 + (rayDirX / rayDirY) * (rayDirX / rayDirY));
37
38 // Case à vérifier
39 int mapCheckX = int(rayStartX + 0.5);
40 int mapCheckY = int(rayStartY + 0.5);
41
42 // Longueur accumulée
43 double rayLengthX = 0.0;
44 double rayLengthY = 0.0;
45
46 // Pas sur chaque composante

48 int stepY = 0;
49
50 // Définir la direction et la distance à l'intersection de la première ligne/colonne

52 stepX = -1;
53 rayLengthX = (0.5 - mapCheckX + rayStartX) * rayUnitStepSizeX;
54 }
55 else {
56 stepX = 1;
57 rayLengthX = (0.5 + mapCheckX - rayStartX) * rayUnitStepSizeX;
58 }
59 if (rayDirY < 0) {
60 stepY = -1;
61 rayLengthY = (0.5 - mapCheckY + rayStartY) * rayUnitStepSizeY;
62 }
63 else {

65 rayLengthY = (0.5 + mapCheckY - rayStartY) * rayUnitStepSizeY;
66 }

74 distance = rayLengthX;
75 rayLengthX += rayUnitStepSizeX;
76 }

79 distance = rayLengthY;
80 rayLengthY += rayUnitStepSizeY;

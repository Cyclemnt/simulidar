16void Map::adjustMapBounds(int amount, Direction dir) {
17 if (amount <= 0) return; // Pas besoin d'extension si amount <= 0
18
19 int mapWidth = robotMap.size();

42// Méthode pour tracer un chemin de cases vides terminé par un mur
43void Map::castRayAndMarkObstacle(double startX, double startY, double rayAngle, double distance) {
44 if (distance == -1) distance = LIDAR_MAX_RANGE;
45
46 // Vecteur directeur du rayon
47 double rayDirX = cos(rayAngle);
48 double rayDirY = sin(rayAngle);
49
50 // Calcul des coordonnées relatives du point de contact du rayon
51 double dx = rayDirX * distance;
52 double dy = rayDirY * distance;
53
54 // Convertir les coordonnées en indices de grille
55 int x0 = std::round(startX);
56 int y0 = std::round(startY);
57 int x1 = std::round(dx + (dx < 0 ? -1 : 1)); // Sur estimer, pour l'extension
58 int y1 = std::round(dy + (dy < 0 ? -1 : 1)); // Sur estimer, pour l'extension
59
60 // Calculs pour les extensions dans chaque direction
61 int colsToAddLeft = -x1 - leftExtension - x0;
62 int colsToAddRight = x1 - robotMap.size() + leftExtension + 1 + x0;
63 int rowsToAddBottom = -y1 - bottomExtension - y0;
64 int rowsToAddTop = y1 - robotMap[0].size() + bottomExtension + 1 + y0;
65
66 // Application des extensions avec la fonction adjustMapBounds
67 adjustMapBounds(colsToAddLeft, Direction::W);
68 adjustMapBounds(colsToAddRight, Direction::E);
69 adjustMapBounds(rowsToAddBottom, Direction::S);
70 adjustMapBounds(rowsToAddTop, Direction::N);
71
72 // Ray Casting identique au fonctionnement du Lidar
73 // Mise à jour de l'état des cases
74
75 int mapWidth = robotMap.size();
76 int mapHeight = robotMap[0].size();
77
78 // Longueur du rayon pour un déplacement unitaire en x ou y
79 double rayUnitStepSizeX = sqrt(1 + (rayDirY / rayDirX) * (rayDirY / rayDirX));
80 double rayUnitStepSizeY = sqrt(1 + (rayDirX / rayDirY) * (rayDirX / rayDirY));
81
82 // Case à vérifier
83 int mapCheckX = floor(startX + 0.5);
84 int mapCheckY = floor(startY + 0.5);
85
86 // Longueur accumulée
87 double rayLengthX = 0.0;
88 double rayLengthY = 0.0;
89
90 // Pas sur chaque composante
91 int stepX = 0;
92 int stepY = 0;
93
94 // Définir la direction et la distance à l'intersection de la première ligne/colonne
95 if (rayDirX < 0) {
96 stepX = -1;
97 rayLengthX = (0.5 - mapCheckX + startX) * rayUnitStepSizeX;
98 }
99 else {
100 stepX = 1;
101 rayLengthX = (0.5 + mapCheckX - startX) * rayUnitStepSizeX;
102 }
103 if (rayDirY < 0) {
104 stepY = -1;
105 rayLengthY = (0.5 - mapCheckY + startY) * rayUnitStepSizeY;
106 }
107 else {
108 stepY = 1;
109 rayLengthY = (0.5 + mapCheckY - startY) * rayUnitStepSizeY;
110 }
111
112 double walkDistance = 0.0;
113 while (walkDistance < (distance - 1e-12)) { // 1e-12 pour contourner les erreurs dues au binaire
114 // Walk
115 if (rayLengthX < rayLengthY) {
116 mapCheckX += stepX;
117 walkDistance = rayLengthX;
118 rayLengthX += rayUnitStepSizeX;
119 }
120 else {
121 mapCheckY += stepY;
122 walkDistance = rayLengthY;
123 rayLengthY += rayUnitStepSizeY;
124 }
125 // Sur le chemin du rayon, placer des cases libres
126 if ((mapCheckX + leftExtension) >= 0 && (mapCheckX + leftExtension) < mapWidth && (mapCheckY + bottomExtension) >= 0 && (mapCheckY + bottomExtension) < mapHeight) {
127 robotMap[mapCheckX + leftExtension][mapCheckY + bottomExtension] = CellState::Free;
128 }
129 }
130 // Si mesure dans les limites, placer un mur à la fin
131 if (distance != LIDAR_MAX_RANGE) {
132 robotMap[mapCheckX + leftExtension][mapCheckY + bottomExtension] = CellState::Wall;
133 }
134}

136// Méthode pour déterminer la prochaine case à explorer
137std::pair<int, int> Map::findNearestInterestPoint(double startX, double startY) const {
138 int ShortestManhattanDistance = std::numeric_limits<int>::max();
139 std::pair<int, int> interestPoint(-1, -1);
140 startX += leftExtension;
141 startY += bottomExtension;
142 // Calcul de la case inconue la plus proche adjacente à une case libre, en distance de Manhattan
143 for (int y = robotMap[0].size() - 1; y >= 0; y--) {
144 for (int x = 0; x < robotMap.size(); x++) {
145 if (robotMap[x][y] == CellState::Unknown) {
146 for (int i = 0; i < 4; i++) {
147 int dx = ((i + 1) % 2) * (i - 1);
148 int dy = (i % 2) * (i - 2);
149 if (x + dx >= 0 && x + dx < robotMap.size() && y + dy >= 0 && y + dy < robotMap[0].size()) {
150 if (robotMap[x + dx][y + dy] == CellState::Free) {
151 if ((abs(startX - x) + abs(startY - y)) < ShortestManhattanDistance) {
152 ShortestManhattanDistance = (abs(startX - x) + abs(startY - y));
153 interestPoint.first = x;
154 interestPoint.second = y;
155 }
156 }
157 }
158 }
159 }
160 }
161 }
162 return interestPoint;
163}

165// Algorithme A* pour trouver le chemin le plus court entre "start" et "goal"
166std::vector<std::pair<int, int>> Map::aStar(std::pair<int, int> start, std::pair<int, int> goal) const {
167 // Algorithme de recherche A* =================================
168 // https://en.wikipedia.org/wiki/A*_search_algorithm#Pseudocode
169
170 int width = robotMap.size(); // Largeur de la carte (nombre de colonnes)
171 int height = robotMap[0].size(); // Hauteur de la carte (nombre de lignes)
172
173 // Ensemble des nœuds à explorer, initialisé avec le nœud de départ
174 std::set<std::pair<int, int>> openSet = {start};
175
176 // Dictionnaire pour conserver le chemin le plus efficace trouvé pour chaque nœud
177 std::map<std::pair<int, int>, std::pair<int, int>> cameFrom;
178
179 // Initialisation de gScore avec Infinity pour tous les nœuds sauf le départ
180 std::map<std::pair<int, int>, double> gScore;
181 for (int x = 0; x < width; ++x) {
182 for (int y = 0; y < height; ++y) {
183 gScore[{x, y}] = std::numeric_limits<double>::infinity(); // Coût infini pour les nœuds non encore explorés
184 }
185 }
186 gScore[start] = 0; // Coût de départ est zéro pour le point de départ
187
188 // Initialisation de fScore avec Infinity, sauf pour le départ où fScore est la distance de Manhattan jusqu'au but
189 std::map<std::pair<int, int>, double> fScore;
190 fScore[start] = std::abs(start.first - goal.first) + std::abs(start.second - goal.second);
191
192 // File de priorité pour les nœuds à explorer, triée par fScore (priorité aux nœuds avec le coût estimé le plus bas)
193 auto cmp = [&fScore](std::pair<int, int> left, std::pair<int, int> right) { return fScore[left] > fScore[right]; };
194 std::priority_queue<std::pair<int, int>, std::vector<std::pair<int, int>>, decltype(cmp)> openQueue(cmp);
195 openQueue.push(start);
196
197 while (!openQueue.empty()) {
198 // Récupère le nœud avec le coût estimé (fScore) le plus bas
199 std::pair<int, int> current = openQueue.top();
200 openQueue.pop();
201
202 // Si on atteint le but, reconstruire et retourner le chemin
203 if (current == goal) {
204 std::vector<std::pair<int, int>> totalPath = {current};
205 while (cameFrom.find(current) != cameFrom.end()) {
206 current = cameFrom[current];
207 totalPath.insert(totalPath.begin(), current); // Insère au début pour obtenir l'ordre de départ vers l'arrivée
208 }
209 return totalPath; // Retourne le chemin complet
210 }
211
212 // Retire le nœud actuel de openSet, car il a été exploré
213 openSet.erase(current);
214
215 // Génère les voisins orthogonaux (haut, bas, gauche, droite)
216 std::vector<std::pair<int, int>> neighbors = {
217 {current.first + 1, current.second},
218 {current.first - 1, current.second},
219 {current.first, current.second + 1},
220 {current.first, current.second - 1}
221 };
222
223 // Parcourt chaque voisin du nœud actuel
224 for (std::pair<int, int> neighbor : neighbors) {
225 // Vérifie si le voisin est dans les limites de la grille
226 if (neighbor.first < 0 || neighbor.first >= width || neighbor.second < 0 || neighbor.second >= height) {
227 continue; // Ignore le voisin s'il est hors des limites
228 }
229
230 // Ignore le voisin s'il est un mur
231 if (robotMap[neighbor.first][neighbor.second] == CellState::Wall) {
232 continue;
233 }
234
235 // Calcule le gScore provisoire du voisin (coût depuis le départ jusqu'à ce voisin)
236 double tentative_gScore = gScore[current] + 1; // Coût de déplacement = 1 pour les déplacements orthogonaux
237
238 // Si le nouveau chemin est plus court ou si le voisin n'a jamais été visité
239 if (tentative_gScore < gScore[neighbor]) {
240 // Met à jour le chemin optimal pour atteindre le voisin
241 cameFrom[neighbor] = current;
242
243 // Met à jour le gScore et le fScore pour le voisin
244 gScore[neighbor] = tentative_gScore;
245 fScore[neighbor] = tentative_gScore + std::abs(neighbor.first - goal.first) + std::abs(neighbor.second - goal.second);
246
247 // Ajoute le voisin à openQueue s'il n'est pas déjà dans openSet
248 if (openSet.find(neighbor) == openSet.end()) {
249 openSet.insert(neighbor);
250 openQueue.push(neighbor);
251 }
252 }
253 }
254 }
255
256 // Retourne un chemin vide si l'objectif n'est pas atteint
257 return {};
258}

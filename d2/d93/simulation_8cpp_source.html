10// Méthode pour démarrer la simulation
11void Simulation::run() {
12 // Instanciation des objets
13 environment = new Environment(ENV_WIDTH, ENV_HEIGHT);
14 map = new Map();
15 robot = new Robot();
16 lidar = new Lidar(this); // Lidar reçoit un pointeur vers Simulation
17 // Assignation des objets au robot
18 robot->setLidar(lidar);
19 robot->setMap(map);
20 robot->setTimeStep(timeStep);
21
22 // LOGIQUE DE SIMULATION
23 environment->generateRandomObstacles(OBSTACLE_NUM, OBSTACLE_MAX_SIZE);
24 initializeRobotPose();
25
26 displayAll();
27 cv::waitKey(0);
28
29 // BOUCLE PRINCIPALE
30 for (int i = 0; i < 10000; ++i) {
31 // Lecture du Lidar
32 std::vector<double> lidarReadings = lidar->readAll();
33 robot->updateMap(lidarReadings);
34 // Chercher un chemin
35 std::pair<int, int> intrstPt = map->findNearestInterestPoint(robot->getX(), robot->getY());
36 int robotMapX = map->getLeftExtension() + floor(robot->getX() + 0.5);
37 int robotMapY = map->getBottomExtension() + floor(robot->getY() + 0.5);
38 std::vector<std::pair<int, int>> path = map->aStar({robotMapX, robotMapY}, intrstPt);
39
40 // Si tout exploré, fin de la simulation
41 if (path.empty()) {
42 std::cout << "Exploration terminée" << std::endl;
43 displayAll();
44 cv::waitKey(0);
45 exit(0);
46 }
47 // Exécuter la première instruction
48 bool stepDone = false;
49 while (!stepDone) {
50 stepDone = robot->executeInstruction(path[1]);
51 displayAll();
52 cv::waitKey(timeStep);
53 }
54 }
55}

57// Méthode pour initialiser la pose du robot dans l'environnement
58void Simulation::initializeRobotPose() {
59 int width = environment->getWidth();
60 int height = environment->getHeight();
61
62 srand(static_cast<unsigned>(time(0)));
63
64 int x, y;
65 bool isValidPosition = false;
66 const int maxAttempts = 1000; // Limite de tentatives pour trouver une position
67 int attempts = 0;
68
69 while (attempts < maxAttempts && !isValidPosition) {
70 attempts++;
71
72 // Génère une position aléatoire
73 x = rand() % width;
74 y = rand() % height;
75
76 // Vérifie que la position est libre pour le robot
77 isValidPosition = (environment->getRoom()[x][y] == CellState::Free);
78 }
79
80 // Vérifie si une position valide a été trouvée
81 if (isValidPosition) {
82 // Place le centre du robot dans la cellule libre trouvée
83 xRobotStart = x;
84 yRobotStart = y;
85 } else {
86 // Affiche une erreur si aucune position valide n'est trouvée
87 throw std::runtime_error("Erreur : Impossible de positionner le robot dans un espace libre.");
88 }
89}

91// Méthode pour afficher la simulation
92void Simulation::displaySimulation(int scaleFactor, Grid plan) const {
93 int height = 0, width = 0;
94 double positionRobotX = 0, positionRobotY = 0, robotOrientation = 0;
95 std::string WindowName = "";
96 if (plan == environment->getRoom()) {
97 height = environment->getHeight();
98 width = environment->getWidth();
99 WindowName = "Environnement";
100 positionRobotX = xRobotStart + robot->getX() + 0.5;
101 positionRobotY = height - yRobotStart - robot->getY() - 0.5;
102 robotOrientation = robot->getOrientation() + orientationRobotStart;
103 }
104 else if (plan == map->getRobotMap()) {
105 height = map->getHeight();
106 width = map->getWidth();
107 WindowName = "Carte du robot";
108 positionRobotX = map->getLeftExtension() + robot->getX() + 0.5;
109 positionRobotY = height - map->getBottomExtension() - robot->getY() - 0.5;
110 robotOrientation = robot->getOrientation();
111 }
112 else {
113 throw std::invalid_argument("Erreur : Impossible d'afficher le plan demandé");
114 }
115 // Création de l'image OpenCV (CV_8UC3 pour une image couleur)
116 cv::Mat roomImage(height * scaleFactor, width * scaleFactor, CV_8UC3);
117 // Dessin de la salle agrandie
118 for (int y = height - 1; y >= 0; y--) {
119 for (int x = 0; x < width; x++) {
120 cv::Rect cell(x * scaleFactor, (height - 1 - y) * scaleFactor, scaleFactor, scaleFactor);
121 switch (plan[x][y]) {
122 case CellState::Wall: cv::rectangle(roomImage, cell, cv::Scalar(255, 0, 0), cv::FILLED); // Murs en bleu
123 break;
124 case CellState::Free: cv::rectangle(roomImage, cell, cv::Scalar(255, 255, 255), cv::FILLED); // Sol en blanc
125 break;
126 case CellState::Unknown: cv::rectangle(roomImage, cell, cv::Scalar(0, 0, 0), cv::FILLED); // Case inconnues en noir
127 break;
128 default: cv::rectangle(roomImage, cell, cv::Scalar(247, 0, 248), cv::FILLED); // Cases non définies en rose
129 break;
130 }
131 }
132 }
133 // Afficher le robot
134 double robotRadius = scaleFactor / 2;
135 cv::Point robotPosition(positionRobotX * scaleFactor, positionRobotY * scaleFactor);
136 cv::circle(roomImage, robotPosition, robotRadius, cv::Scalar(0, 166, 255), cv::FILLED); // Point orange pour le robot
137 // Ligne pour l'orientation
138 cv::line(roomImage, robotPosition, robotPosition + cv::Point(robotRadius * cos(robotOrientation), -robotRadius * sin(robotOrientation)), cv::Scalar(0, 122, 190), 3);
139
140 // Afficher l'image
141 cv::namedWindow(WindowName, cv::WINDOW_GUI_NORMAL);
142 cv::imshow(WindowName, roomImage);
143}

145//Affichage de la vue du robot en 3D avec le raycasting
146void Simulation::displayRaycasting(Grid plan, int WindowWidth, int WindowHeight, int wallHeight, int fov) const {
147 cv::Mat raycastingRender = cv::Mat::zeros(WindowHeight, WindowWidth, CV_8UC3);
148 std::vector<double> readings = lidar->readAll();
149 for (int i = -fov / 2; i <= fov / 2; i++) {
150 double correctedDistance = readings[i + lidar->getRayCount() / 2] * cos(i * M_PI / 180);
151 double projectionWallHeight = wallHeight / correctedDistance;
152 if (readings[i + lidar->getRayCount() / 2] != -1) {
153 double stripeStartY = (WindowHeight - projectionWallHeight) / 2.0;
154 double stripeX = (i + (fov / 2)) * (WindowWidth) / fov;
155 cv::rectangle(raycastingRender, cv::Point(WindowWidth - stripeX, stripeStartY), cv::Point(WindowWidth - (stripeX + (WindowWidth / fov)), stripeStartY + projectionWallHeight), cv::Scalar(255 * (1 - correctedDistance / 30), 0, 0), cv::FILLED);
156 }}
157 cv::imshow("Lidar Raycasting", raycastingRender);
158}

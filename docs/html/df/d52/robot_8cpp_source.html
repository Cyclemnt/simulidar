25void Robot::updateMap(std::vector<double> lidarMeasurements) {
26 for (int i = 0; i < lidar->getRayCount(); i++) {
27 double distance = lidarMeasurements[i];
28 // Calculer l'angle du rayon relatif à l'orientation du robot
29 double rayAngle = orientation + (i - lidar->getRayCount() / 2) * (M_PI / 180.0);
30 // NE PAS PRENDRE LES DIAGONALES ------------+
31 // car l'erreur de la représentation binaire |
32 // des virgules flottantes est plus impactante
33 if (fabs(fmod(fabs(rayAngle), M_PI_2) - M_PI_4) < (0.034)) continue;
34 // ------------------------------------------+
35 // Ray casting dans la grille pour adapter l'état des cases
36 map->castRayAndMarkObstacle(x, y, rayAngle, distance);

41bool Robot::executeInstruction(std::pair<int, int> targetPos_) {
42 if (targetPos != targetPos_) targetPos = targetPos_;
43 // Case actuelle du robot
44 int mapX = floor(x + 0.5) + map->getLeftExtension();
45 int mapY = floor(y + 0.5) + map->getBottomExtension();
46 // Position actuelle du robot sur sa carte
47 double robotX = x + map->getLeftExtension();
48 double robotY = y + map->getBottomExtension();
49
50 // Chemin à parcourir
51 int dx = targetPos.first - mapX;
52 int dy = targetPos.second - mapY;
53
54 bool achievedTarget = false;
55
56 // Détermine l'orientation cible en fonction de dx et dy
57 double targetOrientation = orientation;
58 if (dx > 0) targetOrientation = 0; // Droite
59 else if (dx < 0) targetOrientation = M_PI; // Gauche
60 else if (dy > 0) targetOrientation = M_PI / 2.0; // Haut
61 else if (dy < 0) targetOrientation = 3 * M_PI / 2.0; // Bas
62
63 double angleDiff = targetOrientation - orientation;
64 // Ajuste la différence d'angle pour être dans [-π, π]
65 if (angleDiff > M_PI) angleDiff -= 2 * M_PI;
66 if (angleDiff < -M_PI) angleDiff += 2 * M_PI;
67
68 if (angleDiff == 0) { // Si l'orientation est bonne, avancer
69 double distance = fabs(targetPos.first - robotX) + fabs(targetPos.second - robotY);
70 move(std::min(distance, SPEED * timeStep));
71 } else { // Sinon, tourner
72 rotate(std::clamp(angleDiff, -ANG_V * timeStep, ANG_V * timeStep));
73 }
74 // Recalcul de la position sur la carte
75 robotX = x + map->getLeftExtension();
76 robotY = y + map->getBottomExtension();
77 // Si correspondance avec l'objectif, alors fin
78 if (targetPos.first == robotX && targetPos.second == robotY) achievedTarget = true;
79
80 return achievedTarget;
81}
